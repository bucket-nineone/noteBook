### 5.处理父子项,tree

```js
export function toTree(node) {
  node.forEach(function (it) {
    delete it.children
  })
  // 定义map/
  var map = {}
  // 这里可以重构数据类型，放回字段值
  node.forEach(function (it) {
    map[it.menuId] = it
  })

  // 定义返回集合
  var val = []
  node.forEach(function (it) {
    var parent = map[it.parentId]
    if (parent) {
      // 有数据说明不是顶级节点，将数据放到该 children 子节点下
      ;(parent.children || (parent.children = [])).push(it)
    } else {
      // 没有数据说明是顶级节点放到val中
      val.push(it)
    }
  })
  return val
}
```

export function listToTreedata(list, resArr) {
  list.forEach((item) => {
    if (!resArr.includes(item)) {
      resArr.push(item);
    }
    if (item.children && item.children.length > 0) {
      listToTreedata(item.children, resArr);
    }
  });
  return resArr;
}



//处理树形
export function toTree(node) {
  node.forEach(function (it) {
    delete it.children
  })
  // 定义map/
  var map = {}
  // 这里可以重构数据类型，放回字段值
  node.forEach(function (it) {
    map[it.id] = it
  })

  // 定义返回集合
  var val = []
  node.forEach(function (it) {
    var parent = map[it.parentId]
    if (parent) {
      // 有数据说明不是顶级节点，将数据放到该 children 子节点下
      ;(parent.children || (parent.children = [])).push(it)
    } else {

    }
  })
